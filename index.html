function drawS1Logo() {

  background(0);

  let elapsed = frameCount - stateStartFrame;

  

  // --- 名字淡入控制 ---

  if (elapsed > 30 && !namesStarted) {

    namesStarted = true;

  }

  if (namesStarted && nameFadeIn < 1.0) {

    nameFadeIn += 0.025;

    if (nameFadeIn > 1.0) nameFadeIn = 1.0;

  }

  

  // --- 名字消散控制 ---

  if (elapsed > 180) {

    nameFadeOut = map(elapsed, 180, 220, 1.0, 0);

    if (nameFadeOut < 0) nameFadeOut = 0;

  } else {

    nameFadeOut = 1.0;

  }

  

  let currentNameAlpha = nameFadeIn * nameFadeOut * nameAlpha;


  // --- 滚动名字背景 ---

  let y1 = height * 0.25;

  let y2 = height * 0.75;

  let totalWidth1 = nameSpacing * namesRow1.length;

  let totalWidth2 = nameSpacing * namesRow2.length;


  // 上排：左 -> 右（带高度差）

  let offset1 = (frameCount * 1.2) % totalWidth1;

  fill(255, 200, 100, currentNameAlpha);

  textSize(20);

  textAlign(LEFT, CENTER);

  for (let i = 0; i < namesRow1.length; i++) {

    let baseX = -nameSpacing + i * nameSpacing + offset1;

    let thisY = y1 + row1Offsets[i];

    for (let k = -2; k <= 2; k++) {

      text(namesRow1[i], baseX + k * totalWidth1, thisY);

    }

  }


  // 下排：右 -> 左

  let offset2 = (frameCount * 0.8) % totalWidth2;

  fill(255, 200, 100, currentNameAlpha * 0.85);

  textSize(18);

  for (let i = 0; i < namesRow2.length; i++) {

    let baseX = width + nameSpacing * i - offset2;

    for (let k = -2; k <= 2; k++) {

      text(namesRow2[i], baseX + k * totalWidth2, y2);

    }

  }


  // --- 光点/光线/标题动画 ---

  let breathe = sin(frameCount * 0.025) * 2;


  if (elapsed < 60) {

    logoAlpha = map(elapsed, 0, 30, 0, 255);

    logoRadius = map(elapsed, 0, 60, 0, 8);

    fill(255, 220, 150, logoAlpha);

    noStroke();

    ellipse(width / 2, height / 2, logoRadius * 2, logoRadius * 2);

  } else if (elapsed < 120) {

    logoAlpha = 255;

    logoRadius = 8;

    logoLineLength = map(elapsed, 60, 120, 0, 300);

    fill(255, 220, 150, logoAlpha);

    noStroke();

    ellipse(width / 2, height / 2, logoRadius * 2, logoRadius * 2);

    stroke(255, 220, 150, logoAlpha);

    strokeWeight(2);

    line(width / 2 - logoLineLength, height / 2, width / 2 + logoLineLength, height / 2);

    noStroke();

  } else if (elapsed < 200) {

    logoLineLength = map(elapsed, 120, 180, 300, 0);

    logoTextAlpha = map(elapsed, 150, 200, 0, 255);

    logoAlpha = map(elapsed, 180, 200, 255, 0);

    

    if (logoAlpha > 0) {

      stroke(255, 220, 150, logoAlpha);

      strokeWeight(2);

      line(width / 2 - logoLineLength, height / 2, width / 2 + logoLineLength, height / 2);

      noStroke();

    }

    fill(255, 200, 100, logoTextAlpha);

    textAlign(CENTER, CENTER);

    textSize(88 + breathe);

    text("造梦者", width / 2, height / 2);

  } else {

    fill(255, 200, 100, 255);

    textAlign(CENTER, CENTER);

    textSize(88 + breathe);

    text("造梦者", width / 2, height / 2);

    

    if (elapsed > 240) {

      state = S1_TITLE;

      stateStartFrame = frameCount;

      nameFadeIn = 0;

      nameFadeOut = 1.0;

      namesStarted = false;

    }

  }

}

function drawS1Text() {

  tint(255, 255);

  image(scene1Img, 0, 0, width, height);

  noTint();

  

  for (let i = 0; i < textIndex; i++) {

    drawS1SubtitleFull(lines1[i], 255, lineY1[i]);

  }

  

  if (textIndex < lines1.length) {

    if (charIndex < lines1[textIndex].length) {

      charTimer++;

      if (charTimer > 2) { charIndex++; charTimer = 0; }

      let partialText = lines1[textIndex].substring(0, charIndex);

      drawS1SubtitleFull(partialText, 255, lineY1[textIndex]);

    } else {

      drawS1SubtitleFull(lines1[textIndex], 255, lineY1[textIndex]);

      lineDelayTimer++;

      if (lineDelayTimer > 30) { textIndex++; charIndex = 0; charTimer = 0; lineDelayTimer = 0; if (textIndex >= lines1.length) { state = S1_WAIT; stateStartFrame = frameCount; btnAlpha = 0; btnBreath = 0; } }

    }

  }

}

function drawBreathingButton(x, y, alpha) {

  push();

  translate(x, y);

  scale(btnScale);

  let ringAngle = frameCount * 0.03;

  noFill();

  stroke(255, 200, 100, alpha * 0.35);

  strokeWeight(2.5);

  strokeCap(SQUARE);

  arc(0, 0, 85, 85, ringAngle, ringAngle + PI * 1.2);

  arc(0, 0, 85, 85, ringAngle + PI, ringAngle + PI * 2.2);

  noFill();

  stroke(255, 200, 100, alpha * 0.4);

  strokeWeight(3);

  let haloSize = 72 + sin(btnBreath) * 12;

  ellipse(0, 0, haloSize, haloSize);

  fill(255, 180, 60, alpha);

  noStroke();

  ellipse(0, 0, 48, 48);

  fill(255, 255, 220, alpha);

  ellipse(-8, -8, 14, 14);

  pop();

}

function drawS2Interact() {

  tint(255, 255);

  image(scene2Bg, 0, 0, width, height);

  noTint();

  

  let oldBoatX = boatX;

  let easing = 0.08;

  boatX += (boatTargetX - boatX) * easing;

  boatSpeed = abs(boatX - oldBoatX);

  prevBoatX = oldBoatX;

  

  if (boatSpeed > 0.3) {

    let spawnX = boatX - boatDrawW / 2 + random(-5, 5);

    let spawnY = boatY + boatDrawH / 2 + random(-8, 8);

    ripples.push(new WaterRipple(spawnX, spawnY));

    if (boatSpeed > 2.0) {

      ripples.push(new WaterRipple(spawnX + random(-20, 20), spawnY + random(-5, 5)));

    }

  }

  

  for (let i = ripples.length - 1; i >= 0; i--) {

    let r = ripples[i];

    r.update();

    r.display();

    if (r.isDead()) { ripples.splice(i, 1); }

  }

  

  imageMode(CENTER);

  image(scene2Boat, boatX, boatY, boatDrawW, boatDrawH);

  imageMode(CORNER);

  

  for (let i = 0; i < 2; i++) { drawScene2Subtitle(lines2[i], 180, i); }

function drawS5Outro() {

  if (outroPhase === 0) {

    if (scene4Snapshot != null) { image(scene4Snapshot, 0, 0); }

    outroBlackAlpha = min(outroBlackAlpha + 3, 255);

    fill(0, outroBlackAlpha);

    noStroke();

    if (outroBlackAlpha >= 255) { outroPhase = 1; outroTextIndex = 0; outroTextAlpha = 0; outroTextLineDone = false; outroTextDelay = 0; }

    return;

  }

  background(0);

  switch (outroPhase) {

    case 1:

      for (let i = 0; i < outroTextIndex; i++) { drawOutroTextLine(outroLines[i], 255, outroTextY[i]); }

      if (outroTextIndex < 3) {

        if (!outroTextLineDone) {

          outroTextAlpha = min(outroTextAlpha + 8, 255);

          drawOutroTextLine(outroLines[outroTextIndex], outroTextAlpha, outroTextY[outroTextIndex]);

          if (outroTextAlpha >= 255) { outroTextLineDone = true; outroTextDelay = 0; }

        } else {

          drawOutroTextLine(outroLines[outroTextIndex], 255, outroTextY[outroTextIndex]);

          outroTextDelay++;

          if (outroTextDelay > 90) { outroTextIndex++; outroTextAlpha = 0; outroTextLineDone = false; outroTextDelay = 0; if (outroTextIndex >= 3) { outroPhase = 2; outroTextAlpha = 255; } }

        }

      }

      break;

    case 2:

      outroTextAlpha = max(outroTextAlpha - 2, 0);

      for (let i = 0; i < 3; i++) { drawOutroTextLine(outroLines[i], outroTextAlpha, outroTextY[i]); }

      if (outroTextAlpha <= 0) { noLoop(); }

      break;

  }

}

function mousePressed() {

  if (state === S1_WAIT) {

    let d = dist(mouseX, mouseY, btnX, btnY);

    if (d < 60) { btnScale = 0.8; state = S1_TRANS; stateStartFrame = frameCount; transitionRadius = 0; s1TransBlack = 0; }

  }

  if (state === S3_SCROLL) {

    if (s3BtnVisible) {

      let d = dist(mouseX, mouseY, s3BtnX, s3BtnY);

      if (d < 40) { s3BtnScale = 0.8; s3BtnVisible = false; state = S3_TRANS; stateStartFrame = frameCount; s3TransRadius = 0; s3TransX = s3BtnX; s3TransY = s3BtnY; s3TransBlack = 0; return; }

    }

    for (let i = 0; i < filmFrames.length; i++) {

      let f = filmFrames[i];

      if (mouseX > f.x && mouseX < f.x + f.w && mouseY > f.y && mouseY < f.y + f.h) { s3Paused = true; s3Selected = i; s3ZoomProgress = 0; state = S3_ZOOM; break; }

    }

  } else if (state === S3_ZOOM) { s3UnzoomProgress = 0; state = S3_UNZOOM; }

  if (state === S4_INTERACT && buildings.length < 10) {

    let index = nextBuildingIndex();

    let xPos = s4PresetX[buildings.length];

    let newB = new Building(index, xPos);

    buildings.push(newB);

  }

}
